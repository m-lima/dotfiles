# Initialize container
docker run -v ~/code/SyncTube/src/SyncTube/data:/data -d --name sync -p 80:80 sync

# Update container files (copy files)
docker cp . sync:/app

# Access running container
docker exec -t -i sync /bin/bash

# Start/stop existing container
docker start/stop [container]

# Initialize image
docker build -t sync .

# List all containers
docker ps -a

# Delete container
docker rm [container]

# Delete image
docker rmi [image]
